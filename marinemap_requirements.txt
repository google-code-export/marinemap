# -----------------------------------------
# # Create a virtualenv
# cd ~
# virtualenv MyMarineMap
#
# # Install all requirements
# pip -E MyMarineMap install -r marinemap_requirements.txt
#
# # Activate virtual env
# cd MyMarineMap
# source bin/activate
# # .... do stuff .....
# deactivate
# -----------------------------------------
#
# Note that there are several hairy requirements that aren't handled 
# by this file; these must be installed globally the hard way:
#
# GDAL : just copy the gdal.pth file into your local site packages dir
# Mapnik : just copy the entire mapnik dir into your local site packages dir
# PIL : symlink the PIL directory to site-packages dir
# Postgis : set up according to the marinemap docs
